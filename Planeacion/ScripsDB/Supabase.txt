-- seguridad.roles (corregido: solo una vez)
CREATE TABLE seguridad.roles (
    IdRol INT IDENTITY PRIMARY KEY,
    NombreRol VARCHAR(50) NOT NULL UNIQUE,
    Descripcion VARCHAR(200),
    Activo BIT NOT NULL DEFAULT 1
);
GO

-- seguridad.usuarios
CREATE TABLE seguridad.usuarios (
    IdUsuario INT IDENTITY PRIMARY KEY,
    PrimerNombre VARCHAR(100) NOT NULL,
    SegundoNombre VARCHAR(100),
    PrimerApellido VARCHAR(100) NOT NULL,
    SegundoApellido VARCHAR(100),
    Correo VARCHAR(150) NOT NULL UNIQUE,
    Telefono VARCHAR(20),
    PasswordHash VARCHAR(255) NOT NULL,
    IdRol INT NOT NULL,
    FechaCreacion DATETIME NOT NULL DEFAULT GETDATE(),
    Activo BIT NOT NULL DEFAULT 1,

    CONSTRAINT FK_Usuario_Rol
        FOREIGN KEY (IdRol) REFERENCES seguridad.roles(IdRol),

    -- Check básico (luego agregamos uno más estricto)
    CONSTRAINT CK_Correo_Formato
        CHECK (Correo LIKE '%@%.%')
);
GO

-- academico.categorias
CREATE TABLE academico.categorias (
    IdCategoria INT IDENTITY PRIMARY KEY,
    NombreCategoria VARCHAR(100) NOT NULL UNIQUE,
    EdadMin INT NOT NULL,
    EdadMax INT NOT NULL,

    CONSTRAINT CK_Edades CHECK (EdadMin >= 4 AND EdadMax <= 25 AND EdadMin < EdadMax)
);
GO

-- academico.posiciones
CREATE TABLE academico.posiciones (
    IdPosicion INT IDENTITY PRIMARY KEY,
    NombrePosicion VARCHAR(50) NOT NULL UNIQUE
);
GO

-- academico.temporadas
CREATE TABLE academico.temporadas (
    IdTemporada INT IDENTITY PRIMARY KEY,
    AnioInicio INT NOT NULL,
    AnioFin INT NOT NULL,
    Activa BIT NOT NULL DEFAULT 1,

    CONSTRAINT CK_Temporada CHECK (AnioFin = AnioInicio + 1)
);
GO

-- personal.jugadores
CREATE TABLE personal.jugadores (
    IdJugador INT IDENTITY PRIMARY KEY,
    IdUsuario INT NOT NULL UNIQUE,
    IdCategoria INT NOT NULL,
    IdPosicion INT NOT NULL,
    FechaIngreso DATE NOT NULL DEFAULT GETDATE(),
    Activo BIT NOT NULL DEFAULT 1,

    FOREIGN KEY (IdUsuario) REFERENCES seguridad.usuarios(IdUsuario),
    FOREIGN KEY (IdCategoria) REFERENCES academico.categorias(IdCategoria),
    FOREIGN KEY (IdPosicion) REFERENCES academico.posiciones(IdPosicion)
);
GO

-- personal.entrenadores
CREATE TABLE personal.entrenadores (
    IdEntrenador INT IDENTITY PRIMARY KEY,
    IdUsuario INT NOT NULL UNIQUE,
    Especialidad VARCHAR(100),
    FechaContratacion DATE NOT NULL,
    Activo BIT NOT NULL DEFAULT 1,

    FOREIGN KEY (IdUsuario) REFERENCES seguridad.usuarios(IdUsuario)
);
GO

-- competencia.equipos
CREATE TABLE competencia.equipos (
    IdEquipo INT IDENTITY PRIMARY KEY,
    Nombre VARCHAR(100) NOT NULL,
    IdCategoria INT NOT NULL,
    FechaCreacion DATE NOT NULL DEFAULT GETDATE(),
    Activo BIT NOT NULL DEFAULT 1,

    CONSTRAINT UQ_Equipo UNIQUE (Nombre, IdCategoria),

    FOREIGN KEY (IdCategoria) REFERENCES academico.categorias(IdCategoria)
);
GO

-- competencia.jugador_equipo
CREATE TABLE competencia.jugador_equipo (
    IdJugadorEquipo INT IDENTITY PRIMARY KEY,
    IdJugador INT NOT NULL,
    IdEquipo INT NOT NULL,
    FechaIngreso DATE NOT NULL,
    FechaSalida DATE,

    FOREIGN KEY (IdJugador) REFERENCES personal.jugadores(IdJugador),
    FOREIGN KEY (IdEquipo) REFERENCES competencia.equipos(IdEquipo),

    CONSTRAINT CK_Fechas_Equipo CHECK (FechaSalida IS NULL OR FechaSalida > FechaIngreso)
);
GO

-- competencia.torneos
CREATE TABLE competencia.torneos (
    IdTorneo INT IDENTITY PRIMARY KEY,
    Nombre VARCHAR(150) NOT NULL,
    Tipo VARCHAR(50) NOT NULL,
    FechaInicio DATE NOT NULL,
    FechaFin DATE NOT NULL,
    IdTemporada INT NOT NULL,

    FOREIGN KEY (IdTemporada) REFERENCES academico.temporadas(IdTemporada),

    CONSTRAINT CK_Fechas_Torneo CHECK (FechaFin > FechaInicio)
);
GO

-- competencia.partidos
CREATE TABLE competencia.partidos (
    IdPartido INT IDENTITY PRIMARY KEY,
    IdTorneo INT NOT NULL,
    Fecha DATE NOT NULL,
    Hora TIME NOT NULL,
    Lugar VARCHAR(150),
    Resultado VARCHAR(20),

    FOREIGN KEY (IdTorneo) REFERENCES competencia.torneos(IdTorneo)
);
GO

-- rendimiento.tipo_entrenamiento
CREATE TABLE rendimiento.tipo_entrenamiento (
    IdTipoEntrenamiento INT IDENTITY PRIMARY KEY,
    Nombre VARCHAR(100) NOT NULL UNIQUE
);
GO

-- rendimiento.entrenamientos
CREATE TABLE rendimiento.entrenamientos (
    IdEntrenamiento INT IDENTITY PRIMARY KEY,
    Fecha DATE NOT NULL,
    HoraInicio TIME NOT NULL,
    HoraFin TIME NOT NULL,
    Lugar VARCHAR(150),
    IdEntrenador INT NOT NULL,
    IdTipoEntrenamiento INT NOT NULL,

    FOREIGN KEY (IdEntrenador) REFERENCES personal.entrenadores(IdEntrenador),
    FOREIGN KEY (IdTipoEntrenamiento) REFERENCES rendimiento.tipo_entrenamiento(IdTipoEntrenamiento),

    CONSTRAINT CK_Horas CHECK (HoraFin > HoraInicio)
);
GO

-- rendimiento.asistencias
CREATE TABLE rendimiento.asistencias (
    IdAsistencia INT IDENTITY PRIMARY KEY,
    IdEntrenamiento INT NOT NULL,
    IdJugador INT NOT NULL,
    Estado VARCHAR(20) NOT NULL,

    FOREIGN KEY (IdEntrenamiento) REFERENCES rendimiento.entrenamientos(IdEntrenamiento),
    FOREIGN KEY (IdJugador) REFERENCES personal.jugadores(IdJugador),

    CONSTRAINT CK_Estado CHECK (Estado IN ('Presente','Ausente','Justificado'))
);
GO

-- rendimiento.estadisticas_jugador
CREATE TABLE rendimiento.estadisticas_jugador (
    IdEstadistica INT IDENTITY PRIMARY KEY,
    IdJugador INT NOT NULL,
    Goles INT NOT NULL DEFAULT 0,
    Asistencias INT NOT NULL DEFAULT 0,
    TarjetasAmarillas INT NOT NULL DEFAULT 0,
    TarjetasRojas INT NOT NULL DEFAULT 0,

    FOREIGN KEY (IdJugador) REFERENCES personal.jugadores(IdJugador),

    CONSTRAINT CK_Estadisticas CHECK (Goles >= 0 AND Asistencias >= 0)
);
GO

-- financiero.conceptos_pago
CREATE TABLE financiero.conceptos_pago (
    IdConcepto INT IDENTITY PRIMARY KEY,
    NombreConcepto VARCHAR(100) NOT NULL UNIQUE,
    MontoBase DECIMAL(10,2) NOT NULL,
    EsRecurrente BIT NOT NULL DEFAULT 0,
    Activo BIT NOT NULL DEFAULT 1,

    CONSTRAINT CK_Monto CHECK (MontoBase > 0)
);
GO

-- financiero.facturas
CREATE TABLE financiero.facturas (
    IdFactura INT IDENTITY PRIMARY KEY,
    IdJugador INT NOT NULL,
    FechaEmision DATE NOT NULL DEFAULT GETDATE(),
    FechaVencimiento DATE NOT NULL,
    Estado VARCHAR(20) NOT NULL DEFAULT 'Pendiente',

    FOREIGN KEY (IdJugador) REFERENCES personal.jugadores(IdJugador),

    CONSTRAINT CK_EstadoFactura CHECK (Estado IN ('Pendiente','Pagada','Vencida')),
    CONSTRAINT CK_Vencimiento CHECK (FechaVencimiento >= FechaEmision)
);
GO

-- financiero.pagos
CREATE TABLE financiero.pagos (
    IdPago INT IDENTITY PRIMARY KEY,
    IdFactura INT NOT NULL,
    FechaPago DATE NOT NULL DEFAULT GETDATE(),
    MontoPagado DECIMAL(10,2) NOT NULL,

    FOREIGN KEY (IdFactura) REFERENCES financiero.facturas(IdFactura),

    CONSTRAINT CK_MontoPago CHECK (MontoPagado > 0)
);
GO

-- Jugadores con más goles que el promedio general

    FROM rendimiento.estadisticas_jugador

-- Equipo con más jugadores inscritos
SELECT e.Nombre
FROM competencia.equipos e
WHERE (
    SELECT COUNT(*)
    FROM competencia.jugador_equipo je
    WHERE je.IdEquipo = e.IdEquipo
) = (
    SELECT MAX(Cantidad)
    FROM (
        SELECT COUNT(*) AS Cantidad
        FROM competencia.jugador_equipo
        GROUP BY IdEquipo
    ) AS Sub
);

-- Entrenadores con más entrenamientos que el promedio
SELECT
    u.PrimerNombre,
    u.PrimerApellido,
    COUNT(en.IdEntrenamiento) AS TotalSesiones
FROM rendimiento.entrenamientos en
INNER JOIN personal.entrenadores e
    ON en.IdEntrenador = e.IdEntrenador
INNER JOIN seguridad.usuarios u
    ON e.IdUsuario = u.IdUsuario
GROUP BY u.PrimerNombre, u.PrimerApellido
HAVING COUNT(en.IdEntrenamiento) > (
    SELECT AVG(Sesiones)
    FROM (
        SELECT COUNT(*) AS Sesiones
        FROM rendimiento.entrenamientos
        GROUP BY IdEntrenador
    ) AS Prom
);

-- Jugadores que han asistido a todos los entrenamientos
SELECT
    u.PrimerNombre,

INNER JOIN seguridad.usuarios u

-- KPI 1: Tasa de Morosidad
SELECT
    COUNT(CASE WHEN Estado = 'Pendiente' THEN 1 END) * 100.0
    / NULLIF(COUNT(*), 0) AS TasaMorosidad
FROM financiero.facturas;

-- KPI 2: Nivel de Recaudo Mensual
SELECT
    YEAR(FechaPago) AS Año,
    MONTH(FechaPago) AS Mes,
    SUM(MontoPagado) AS TotalRecaudado
FROM financiero.pagos
GROUP BY YEAR(FechaPago), MONTH(FechaPago)
ORDER BY Año, Mes;

-- KPI 3: Ticket Promedio por Jugador
SELECT
    SUM(MontoPagado) * 1.0 /
    NULLIF((SELECT COUNT(*) FROM personal.jugadores), 0) AS TicketPromedio
FROM financiero.pagos;

-- KPI 4: Promedio de Goles por Jugador
SELECT AVG(Goles) AS PromedioGoles
FROM rendimiento.estadisticas_jugador;

-- KPI 5: Tasa de Asistencia a Entrenamientos
SELECT
    SUM(CASE WHEN Estado = 'Presente' THEN 1 ELSE 0 END) * 100.0
    / NULLIF(COUNT(*), 0) AS TasaAsistencia
FROM rendimiento.asistencias;

-- KPI 6: Promedio de Goles por Equipo
SELECT
    e.Nombre,
    AVG(est.Goles) AS PromedioGolesEquipo
FROM competencia.equipos e
INNER JOIN competencia.jugador_equipo je
    ON e.IdEquipo = je.IdEquipo
INNER JOIN rendimiento.estadisticas_jugador est
    ON je.IdJugador = est.IdJugador
GROUP BY e.Nombre;

-- KPI 7: Crecimiento de Inscripciones
SELECT
    YEAR(FechaIngreso) AS Año,
    COUNT(*) AS NuevosJugadores
FROM personal.jugadores
GROUP BY YEAR(FechaIngreso);

-- KPI 8: Ratio Entrenador / Jugadores
SELECT
    (SELECT COUNT(*) FROM personal.jugadores) * 1.0 /
    NULLIF((SELECT COUNT(*) FROM personal.entrenadores), 0) AS RatioEntrenadorJugador;

-- OKR 1: Reducir Morosidad al 10%
SELECT
    CASE
        WHEN (
            COUNT(CASE WHEN Estado = 'Pendiente' THEN 1 END) * 100.0
            / NULLIF(COUNT(*), 0)
        ) <= 10 THEN 'Objetivo Cumplido'
        ELSE 'Objetivo NO Cumplido'
    END AS EstadoOKR
FROM financiero.facturas;

-- OKR 2: Lograr Asistencia Superior al 85%
SELECT
    CASE
        WHEN (
            SUM(CASE WHEN Estado = 'Presente' THEN 1 ELSE 0 END) * 100.0
            / NULLIF(COUNT(*), 0)
        ) >= 85 THEN 'Objetivo Cumplido'
        ELSE 'Objetivo NO Cumplido'
    END AS EstadoOKR
FROM rendimiento.asistencias;

-- OKR 3: Tener al menos 2 jugadores con más de 10 goles
SELECT COUNT(*) AS JugadoresElite
FROM rendimiento.estadisticas_jugador
WHERE Goles > 10;

SELECT
    CASE
        WHEN COUNT(*) >= 2 THEN 'Objetivo Cumplido'
        ELSE 'Objetivo NO Cumplido'
    END AS EstadoOKR
FROM rendimiento.estadisticas_jugador
WHERE Goles > 10;

/* ============================================================
   1. VER LOGINS Y SUS MAPPINGS A USUARIOS DE BD
============================================================ */
SELECT
    sp.name             AS Login_Servidor,
    sp.type_desc        AS Tipo_Login,
    dp.name             AS User_BD,
    dp.default_schema_name AS Schema_Defecto
FROM sys.server_principals sp
LEFT JOIN sys.database_principals dp
    ON sp.sid = dp.sid
WHERE sp.name IN ('login_admin','login_entrenador','login_auditor');
GO

/* ============================================================
   2. VER QUÉ ROLES TIENE CADA USUARIO EN LA BD
============================================================ */
SELECT
    dp_user.name    AS Usuario,
    dp_role.name    AS Rol_Asignado
FROM sys.database_role_members drm
INNER JOIN sys.database_principals dp_role
    ON drm.role_principal_id = dp_role.principal_id
INNER JOIN sys.database_principals dp_user
    ON drm.member_principal_id = dp_user.principal_id
WHERE dp_user.name IN ('admin','entrenador','auditor');
GO

/* ============================================================
   3. VER PERMISOS ASIGNADOS A CADA ROL (GRANT / DENY)
============================================================ */
SELECT
    dp.name         AS Rol,
    p.state_desc    AS Tipo_Permiso,   -- GRANT o DENY
    p.permission_name AS Permiso,
    p.class_desc    AS Clase,
    SCHEMA_NAME(major_id) AS Esquema
FROM sys.database_permissions p
INNER JOIN sys.database_principals dp
    ON p.grantee_principal_id = dp.principal_id
WHERE dp.name IN ('rol_admin_total','rol_entrenador_operativo','rol_auditor_readonly')
ORDER BY dp.name, p.state_desc, p.permission_name;
GO
